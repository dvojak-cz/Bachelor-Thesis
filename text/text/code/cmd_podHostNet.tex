# Crete pod with ".spec.hostNetwork=true"
[1]$ kubectl apply -f - <<EOF
apiVersion: v1
kind: Pod
metadata:
  name: my-pod
spec:
  hostNetwork: true
  containers:
    - image: nicolaka/netshoot:v0.9
      name: my-pod
      command: ["tail"]
      args: ["/dev/random"]
EOF

# Attach to newly created pod
[1]$ kubectl exec -it my-pod -- bash
